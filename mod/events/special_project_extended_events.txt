country_event = {
	id = SpecialProjectExtendedEvents.1
	title = ASTEROID_CRASH_HARVESTAGE
	desc = ASTEROID_CRASH_HARVESTAGE_DESC
	picture = GFX_evt_asteroid_approaching_planet
	show_sound = event_super_explosion
	is_triggered_only = yes
	auto_opens = no
	location = event_target:special_project_target

	trigger = {
		any_planet_within_border = { can_asteroid_crash_harvestage = yes }
	}

	immediate = {
		random_planet_within_border = {
			limit = { can_asteroid_crash_harvestage = yes }
			set_timed_planet_flag = { flag = asteroid_crash_harvestage years = 6 }
			save_event_target_as = special_project_target
		}
	}

	option = {
		name = RESEARCH_PROJECT_ORDER
		enable_special_project = {
			name = "ASTEROID_CRASH_HARVESTAGE"
			location = event_target:special_project_target
		}
	}
	option = {
		name = 01_FALLEN_EMPIRE_DISMISSIVE_DOW_SELECT
	}
}
