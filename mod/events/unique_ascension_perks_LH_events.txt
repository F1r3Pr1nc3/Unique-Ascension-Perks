	picture = GFX_evt_system_localization
	show_sound = event_laboratory_sound
	location = from
	is_triggered_only = yes
	auto_select = no
	force_open = no
	# trigger = {
	# 	exists = from
	# 	from = { is_scope_valid = yes is_scope_type = country }
	# }
	immediate = {
		if = {
			limit = { exists = event_target:lh_basic_system }
			export_trigger_value_to_variable = {
				trigger = distance
				parameters = { source = event_target:lh_basic_system max_distance <= 100 type = euclidean }
				variable = basic_distance
			}
			log="Distance from \\[This.GetName] \\[This.basic_distance] to \\[event_target:lh_basic_system.GetName]"
		}
	}
	after = {
		if = {
			limit = { is_variable_set = basic_distance }
			if = {
				limit = { count_neighbor_system = { limit = { } count = 0 } is_variable_set = basic_distance }
				random_neighbor_system_euclidean = {
					limit = {
						NOT = { has_hyperlane_to = prev }
						OR = {
							has_star_flag = lh_spawned_system
							has_star_flag = lh_basic_system
							has_star_flag = lh_special_system
						}
					}
					weights = {
						base = 1
						# modifier = { add = 5 has_star_flag = lh_spawned_system }
						modifier = {
							add = 9
							exists = event_target:lh_basic_system
							check_variable_arithmetic = {
								which = value:system_distance|SOURCE|event_target:lh_basic_system|max_distance|root.basic_distance|
								value < root.basic_distance
							}
						}
					}
					add_hyperlane = { from = this to = prev }
				}
			}
			else_if = {
				limit = {
					exists = event_target:lh_basic_system
					event_target:lh_basic_system = { has_hyperlane_to = prev }
					any_neighbor_system_euclidean = {
						NOT = { has_hyperlane_to = prev }
						OR = {
							has_star_flag = lh_spawned_system
							has_star_flag = lh_basic_system
							has_star_flag = lh_special_system
						}
						check_variable_arithmetic = {
							which = value:system_distance|SOURCE|event_target:lh_basic_system|max_distance|root.basic_distance|
							value < root.basic_distance
						}
						check_variable_arithmetic = {
							which = value:system_distance|SOURCE|root|max_distance|root.basic_distance|
							value < root.basic_distance
						}
					}
				}
				random_neighbor_system_euclidean = {
					limit = {
						NOT = { has_hyperlane_to = prev }
						OR = {
							has_star_flag = lh_spawned_system
							has_star_flag = lh_basic_system
							has_star_flag = lh_special_system
						}
						check_variable_arithmetic = {
							which = value:system_distance|SOURCE|event_target:lh_basic_system|max_distance|root.basic_distance|
							value < root.basic_distance
						}
						check_variable_arithmetic = {
							which = value:system_distance|SOURCE|root|max_distance|root.basic_distance|
							value < root.basic_distance
						}
					}
					# weights = {
					# 	base = 1
					# 	modifier = { add = 25 has_star_flag = lh_spawned_system }
					# }
					remove_hyperlane = { from = event_target:lh_basic_system to = prev }
					add_hyperlane = { from = this to = prev }
				}
			}
			else = {
				random = { chance = 35 }
				random_neighbor_system_euclidean = {
					limit = {
						OR = {
							has_star_flag = lh_spawned_system
							has_star_flag = lh_basic_system
							has_star_flag = lh_special_system
						}
						distance = {
							source = prev
							type = euclidean
							min_distance >= 1
							max_distance <= 25
						}
						check_variable_arithmetic = {
							which = value:system_distance|SOURCE|event_target:lh_basic_system|max_distance|root.basic_distance|
							value < root.basic_distance
						}
						check_variable_arithmetic = {
							which = value:system_distance|SOURCE|root|max_distance|root.basic_distance|
							value < root.basic_distance
						}
					}
					# weights = {
					# 	base = 1
					# 	modifier = { add = 25 has_star_flag = lh_spawned_system }
					# }
					remove_hyperlane = { from = event_target:lh_basic_system to = prev }
					add_hyperlane = { from = this to = prev }
				}
			}
			clear_variable = basic_distance
		}
	}
	option = {
		name = EXCELLENT
		hidden_effect = {
			random_neighbor_system_euclidean = {
				limit = {
					distance = {
						source = prev
						type = euclidean
						min_distance >= 1
						max_distance <= 25
					}
					NOR = {
						exists = owner
						has_hyperlane_to = prev
					}
					OR = {
						has_star_flag = lh_spawned_system
						has_star_flag = lh_basic_system
						has_star_flag = lh_special_system
					}
				}
				add_hyperlane = { from = this to = prev }
			}
		}
	}
}
