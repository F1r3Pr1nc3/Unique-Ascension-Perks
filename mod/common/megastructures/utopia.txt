utopia_site = {
	entity = "construction_platform_entity"
	construction_entity = "construction_platform_entity"
	portrait = "GFX_megastructure_construction_background"
	construction_blocks_others = no
	place_entity_on_planet_plane = no
	entity_offset = { x = -7 y = -7 }
	build_time = 900
	resources = {
		category = megastructures
		cost = { alloys = 5000 influence = 200 }
		upkeep = { energy = 3 }
	}
	potential = { has_ascension_perk = ap_advanced_government_2 }
	possible = {
		exists = starbase
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
	}

	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "requires_no_existing_megastructure"
				# has_any_megastructure = no
				solar_system = { has_no_non_gate_megastructure = yes }
				NOR = {
					has_planet_flag = megastructure
					has_planet_flag = has_megastructure
				}
			}

			custom_tooltip = {
				fail_text = "requires_survey_not_habitable"
				is_surveyed = {			# Prevent leaking habitability information
					who = prev.from
					status = yes
				}
				colonizable_planet = no
			}
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				has_anomaly = no
			}

 			# Balance for think tanks
			custom_tooltip = {
				fail_text = "requires_not_minor_planetary_body"
				NOR = {
					is_asteroid = yes
					is_moon = yes
				}
			}
			custom_tooltip = {
				fail_text = "requires_not_star"
				is_star = no
			}
			custom_tooltip = {
				fail_text = "requires_not_ring_world"
				is_ringworld = no
			}
		} # Use these for all non-star megastructures
	}

	# root = system
	# from = country
	ai_weight = {
		factor = 10
		modifier = {
			factor = 0.1
			starbase = { NOT = { has_starbase_size >= starbase_starfortress } }
		}
		modifier = {
			factor = 0.1
			any_neighbor_system = {
				exists = owner
				NOT = {
					is_owned_by = from
				}
			}
		}
		modifier = {
			factor = 2.0
			from = {
				is_authoritarian = yes
			}
		}
	}
	on_build_start = {}
	on_build_cancel = {}
	on_build_complete = {
		set_star_flag = utopia_built
		fromfrom.planet = { set_planet_flag = has_megastructure }
		from = { country_event = { id = UniqueAscensionPerksEvents.90 } }
	}
}
