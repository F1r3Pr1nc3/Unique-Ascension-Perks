building_foundry_2 = {
	base_buildtime = @b2_time
	category = manufacturing
	can_build = no

	building_sets = {
		industrial
		foundry
	}

	potential = {
		has_owner = yes
		owner = { is_wilderness_empire = no } # see 'building_natural_furnace_2'
		is_special_colony_type = no
	}

	allow = {
		is_occupied_flag = no
	}

	destroy_trigger = {
		has_owner = yes
		OR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
			has_modifier = gestation_colony
			owner = { is_wilderness_empire = yes }
			AND = {
				owner = { is_ai = yes }
				has_any_industry_zone = no
			}
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b2_minerals
			volatile_motes = @b3_rare_cost
		}
		upkeep = {
			energy = @b2_upkeep
			volatile_motes = @b3_rare_upkeep
		}
	}

	prerequisites = {
		tech_alloys_1
	}

	show_tech_unlock_if = {
		is_wilderness_empire = no
	}

	inline_script = {
		script = jobs/foundry_add
		AMOUNT = @building_static_jobs_high
	}

	upgrades = {
		"building_foundry_3"
	}

	planet_modifier = {
		planet_metallurgists_alloys_produces_add = 1
	}

	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = { is_catalytic_empire = no }
		}
		planet_metallurgists_minerals_upkeep_add = 2
	}
	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = { is_gestalt = no is_catalytic_empire = yes }
		}
		planet_metallurgists_food_upkeep_add = 3
		job_catalytic_technician_add = @b1_jobs
	}
	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = { is_hive_empire = yes is_catalytic_empire = no }
		}
		job_alloy_drone_add = @b1_jobs
	}
	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = { is_machine_empire = yes is_catalytic_empire = no }
		}
		job_fabricator_add = @b1_jobs
	}
	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = { is_gestalt = yes is_catalytic_empire = yes }
		}
		planet_metallurgists_food_upkeep_add = 3
		job_catalytic_drone_add = @b1_jobs
	}
	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = {
				OR = {
					has_ascension_perk = ap_pyramidal_society
					has_country_flag = ap_pyramidal_society
				}
				is_common_empire = yes
			}
		}
		job_industrialist_add = 100
	}

	triggered_desc = {
		trigger = {
			has_owner = yes
			owner = { is_gestalt = no is_catalytic_empire = yes }
		}
		text = job_catalytic_technician_effect_desc
	}
	triggered_desc = {
		trigger = {

building_factory_1 = {
	can_build = yes
	base_buildtime = @b1_time
	category = manufacturing

	building_sets = {
		industrial
		factory
	}

	potential = {
		has_owner = yes
		owner = {
			OR = {
				country_uses_consumer_goods = yes
				has_origin = origin_fear_of_the_dark
			}
		}
		NOR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
			has_modifier = gestation_colony
		}
		OR = {
			has_planet_flag = ignore_ai_building_limitations
			owner = {
				OR = {
					is_ai = no
					# Obsessional machines can build these freely. Have fun paperclipping.
					has_valid_civic = civic_machine_obsessional_directive
				}
			}
			# AI is allowed to build one of these if they have a FACTORY specialization - intentionally more restrictive than foundries.
			AND = {
				has_any_industry_zone = yes
				owner = { should_ai_focus_on_trade = no }
			}
		}
	}

	allow = {
		is_occupied_flag = no
	}

	show_tech_unlock_if = {
		OR = {
			country_uses_consumer_goods = yes
			has_origin = origin_fear_of_the_dark
		}
	}

	destroy_trigger = {
		# Vfix sync
		has_owner = yes
		OR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
			has_modifier = gestation_colony
			owner = {
				OR = {
					NOR = {
						country_uses_consumer_goods = yes
						has_origin = origin_fear_of_the_dark
					}
					AND = {
						should_ai_focus_on_trade = yes
						has_monthly_income = { resource = consumer_goods value > 30 }
					}
				}
			}
			AND = {
				owner = { is_ai = yes }
				has_any_industry_zone = no
			}
		}
	}

	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = { is_gestalt = no is_crafter_empire = no }
		}
		job_artisan_add = @b1_jobs
	}
	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = { is_gestalt = no is_crafter_empire = yes }
		}
		job_artificer_add = @b1_jobs
	}
	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = { is_gestalt = yes }
		}
		job_artisan_drone_add = @b1_jobs
	}
	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = {
				OR = {
					has_ascension_perk = ap_pyramidal_society
					has_country_flag = ap_pyramidal_society
				}
				is_common_empire = yes
			}
		}
		job_industrialist_add = @b1_jobs
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
	}

	inline_script = {
		script = jobs/factory_add
		AMOUNT = @building_static_jobs
	}

	prerequisites = {
		tech_basic_industry
	}

	upgrades = {
		"building_factory_2"
	}

	triggered_desc = {
		trigger = {
			has_owner = yes
			owner = { is_gestalt = no is_crafter_empire = no }
		}
		text = job_artisan_effect_desc
	}
	triggered_desc = {
		trigger = {
			has_owner = yes
			owner = { is_gestalt = no is_crafter_empire = yes }
		}
		text = job_artificer_effect_desc
	}
	triggered_desc = {
		trigger = {
			has_owner = yes
			owner = { is_gestalt = yes }
		}
		text = job_artisan_drone_effect_desc
	}
}
