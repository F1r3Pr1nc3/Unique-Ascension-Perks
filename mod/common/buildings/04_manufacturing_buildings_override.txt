		OR = {
			NOT = { exists = owner }

			}
		}

			planet_metallurgists_food_upkeep_add = 3

		}
	}

	triggered_planet_modifier = {
		potential = {

building_factory_1 = {
	base_buildtime = @b1_time
	base_cap_amount = 1
	category = manufacturing

	potential = {
		exists = owner
		owner = {
			OR = {
				country_uses_consumer_goods = yes
				has_origin = origin_fear_of_the_dark
			}
		}
		NOR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
			has_modifier = crucible_colony
		}
		OR = {
			owner = {
				is_ai = no
			}
			has_any_industry_district = yes
		}
	}

	destroy_trigger = {
		OR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
			has_modifier = crucible_colony

			AND = {
				exists = owner
				OR = {
					owner = {
						NOR = {
							country_uses_consumer_goods = yes
							has_origin = origin_fear_of_the_dark

						owner = { is_ai = no }

			}

building_ministry_production = {
	base_buildtime = @b2_time
	base_cap_amount = 1
	icon = building_production_center

	category = manufacturing

	potential = {
		exists = owner
		owner = { is_regular_empire = yes }
		NOR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
		}
		OR = {
			owner = {
				is_ai = no
			}
			has_any_industry_district = yes
		}
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = resort_colony
			has_modifier = slave_colony
			AND = {
				owner = {
					is_ai = yes
				}
				has_any_industry_district = no
			}
		}
	}

	allow = {
		has_major_upgraded_capital = yes
	}

	planet_modifier = {
		job_politician_add = 1
		planet_metallurgists_produces_mult = 0.15
		planet_artisans_produces_mult = 0.15
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				OR = {
					has_ascension_perk = ap_pyramidal_society
					has_country_flag = ap_pyramidal_society
				}
				is_common_empire = yes
			}
		}
		modifier = {
			job_industrialist_add = 1
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b2_minerals
			volatile_motes = @b2_rare_cost
		}
		upkeep = {
			energy = @b2_upkeep
			volatile_motes = @b2_rare_upkeep
		}
	}

	prerequisites = {
		"tech_global_production_strategy"
	}

	triggered_desc = {
		text = job_politician_effect_desc
	}

	show_tech_unlock_if = {
		is_regular_empire = yes
	}

	ai_resource_production = {
		alloys = 6
		trigger = {
			planet_resource_compare = {
				resource = alloys
				value >= 40
				type = produces
			}
		}
	}
	ai_resource_production = {
		consumer_goods = 8
		trigger = {
			planet_resource_compare = {
				resource = consumer_goods
				value >= 50
				type = produces
			}
		}
	}
}
