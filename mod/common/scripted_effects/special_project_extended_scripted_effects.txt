check_special_project_target = {
	if = {
		limit = { NOT = { exists = event_target:special_project_target }}
		log_error = "Event missing special_project_target"
	}
}

special_project_destroy_ship = {
	random_owned_ship = {
		limit = { fleet = { is_same_value = event_target:special_project_fleet } }
		destroy_ship = this
	}
}

create_special_project_amoeba_country = {
	if = {
		limit = { NOT = { exists = event_target:special_project_amoeba_country } }
		create_country = {
			name = "NAME_Spaceborne_Organics"
			type = special_project_aliens
			flag = {
				icon = { category = "zoological" file = "flag_zoological_1.dds" }
				background = { category = "backgrounds" file = "00_solid.dds" }
				colors = { "black" "black" "null" "null" }
			}
		}
		last_created_country = {
			save_global_event_target_as = special_project_amoeba_country
			every_country = { limit = { is_amoeba_country_type = yes } set_faction_hostility = { target = prev set_hostile = no set_neutral = yes set_friendly = no } }
			every_country = { establish_communications_no_message = last_created_country }
		}
	}
}

			}
			give_technology = { tech = tech_titans }
			give_technology = { tech = tech_arc_emitter_1 }
			give_technology = { tech = tech_zero_point_power }
