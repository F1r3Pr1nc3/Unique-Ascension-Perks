expedition_navy_fleet_found = {
	hidden_effect = {
		random_owned_planet = { limit = { is_capital = yes } save_event_target_as = spaw_location }
		create_fleet = {
			name = "Expedition Fleet"
			settings = { spawn_debris = no }
			effect = {
				set_owner = root
				while = { count = 14 random = { chance = 75 create_ship = { name = random design = "Standard Basic Corvette" } } }
				while = { count = 6 random_list = { 15 = { modifier = { factor = 0 NOT = { ROOT = { has_technology = tech_destroyers } } } create_ship = { name = random design = "Standard Basic Destroyer" } } 20 = {} } }
				while = { count = 3 random_list = { 15 = { modifier = { factor = 0 NOT = { ROOT = { has_technology = tech_cruisers } } } create_ship = { name = random design = "Standard Basic Cruiser" } } 20 = {} } }

				set_aggro_range_measure_from = self

		}
	}
