# Returns fallback capital_planet
# This = country
get_capital_planet = {
	if = {
		limit = { exists = capital_scope }
		capital_scope = { save_event_target_as = capital_planet }
	} else_if = {
		limit = { any_owned_planet = { is_colony = yes } }
		random_owned_planet = {
			limit = {
				is_colony = yes
				is_colonizable = no
				exists = space_owner
				solar_system = { exists = starbase }
			}
			weights = {
				base = 1
				modifier = { add = 3 solar_system = { is_capital_system = yes } }
				modifier = { add = 3 num_pops > 30 }
			}
			save_event_target_as = capital_planet
		}
		if = {
			limit = { NOT  = { exists = event_target:capital_planet } }
			random_planet_within_border = {
				limit = {
					has_owner = yes
					is_colony = yes
					OR = {
						AND = { exists = owner is_owned_by = root }
						AND = { exists = controller controller = { is_same_value = root } }
					}
				}
				weights = {
					base = 1
					modifier = { add = 1 exists = owner is_owned_by = root }
					modifier = { add = 1 num_pops > 5 }
				}
				save_event_target_as = capital_planet
			}
		}
	} else = {
		log="Failed: no own planet found for \\[Root.GetName]?!"
	}
}
