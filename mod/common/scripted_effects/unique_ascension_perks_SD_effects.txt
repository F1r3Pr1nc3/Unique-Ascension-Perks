create_domestication_spaceborn_alien_country = {
	save_event_target_as = main_country
	if = {
		limit = { any_relation = { is_country_type = domesticated_spaceborn_alien has_relation_flag = { who = event_target:main_country flag = domesticated_spaceborn_alien } } }
		random_relation = {
			limit = { is_country_type = domesticated_spaceborn_alien has_relation_flag = { who = event_target:main_country flag = domesticated_spaceborn_alien } }
			save_event_target_as = domesticated_spaceborn_alien
		}
	} else = {
		create_country = {
			name = "Domesticated Spaceborn Aliens"
			type = domesticated_spaceborn_alien
			flag = {
				icon = { category = "zoological" file = "flag_zoological_1.dds" }
				background= { category = "backgrounds" file = "00_solid.dds" }
				colors = { "black" "black" "null" "null" }
			}
			effect = {
				every_country = {
					limit = { has_communications = event_target:main_country }
					establish_communications_no_message = prev
				}
				add_modifier = { modifier = domesticated_aliens }
				save_event_target_as = domesticated_spaceborn_alien
				set_country_flag = domesticated_spaceborn_alien
				set_relation_flag = { who = event_target:main_country flag = domesticated_spaceborn_alien }
				set_subject_of = { who = event_target:main_country preset = preset_space_domestication }
				# set_faction_hostility = { target = event_target:main_country set_hostile = no set_neutral = no set_friendly = yes }
				if = { limit = { event_target:main_country = { has_country_flag = space_domestication_tiyanki } } set_country_flag = domesticated_spaceborn_alien_tiyanki } else_if = { limit = { event_target:main_country = { has_country_flag = space_domestication_amoeba } } set_country_flag = domesticated_spaceborn_alien_amoeba } else_if = { limit = { event_target:main_country = { has_country_flag = space_domestication_crystal } } set_country_flag = domesticated_spaceborn_alien_crystal } else_if = { limit = { event_target:main_country = { has_country_flag = space_domestication_cloud } } set_country_flag = domesticated_spaceborn_alien_cloud } else_if = { limit = { event_target:main_country = { has_country_flag = space_domestication_drone } } set_country_flag = domesticated_spaceborn_alien_drone }
			}
		}
	}
	capital_star = { save_event_target_as = spawn_location }
	event_target:domesticated_spaceborn_alien = {
		create_fleet = {
			name = "Domesticated Fertilized Aliens"
			settings = { spawn_debris = no is_boss = yes }
			effect = {
				set_owner = prev
				if = { limit = { event_target:main_country = { has_country_flag = space_domestication_tiyanki } } create_ship = { name = "" design = "Domesticated Mother Tiyanki" } } else_if = { limit = { event_target:main_country = { has_country_flag = space_domestication_amoeba } } while = { count = 25 create_ship = { name = "" design = "Demosticated Amoeba Mother" } } } else_if = { limit = { event_target:main_country = { has_country_flag = space_domestication_crystal } } create_ship = { name = "" design = "Domesticated Mother Crystal" } } else_if = { limit = { event_target:main_country = { has_country_flag = space_domestication_cloud } } create_ship = { name = "" design = "Domesticated Mother Cloud" } } else_if = { limit = { event_target:main_country = { has_country_flag = space_domestication_drone } } create_ship = { name = "" design = "Reprogrammed Ancient Replicator Drone" } }
				set_location = { target = event_target:spawn_location distance = 250 angle = random }
				set_fleet_stance = aggressive
				set_aggro_range_measure_from = self
				set_aggro_range = 5000
				set_formation_scale = 4.0
				save_event_target_as = domesticated_fleet
				set_fleet_flag = domesticated_fleet
				set_fleet_flag = domesticated_mother_fleet
				solar_system = {
					if = { limit = { event_target:main_country = { has_country_flag = space_domestication_tiyanki NOT = { any_planet_within_border = { has_planet_flag = domesticated_alien_home } } } } spawn_planet = { class = pc_gas_giant location = prev has_ring = no init_effect = { set_name = "NAME_TYANKI_HOMEWORLD" set_planet_flag = domesticated_alien_home set_planet_flag = domesticated_alien_home_tiyanki set_planet_size = 50 clear_deposits = yes } } } else_if = { limit = { event_target:main_country = { has_country_flag = space_domestication_amoeba NOT = { any_planet_within_border = { has_planet_flag = domesticated_alien_home_amoeba } } } } spawn_planet = { class = pc_toxic location = prev has_ring = no init_effect = { set_name = "NAME_AMEOBA_HOMEWORLD" set_planet_flag = domesticated_alien_home set_planet_flag = domesticated_alien_home_amoeba set_planet_size = 40 clear_deposits = yes } } } else_if = { limit = { event_target:main_country = { has_country_flag = space_domestication_crystal NOT = { any_planet_within_border = { has_planet_flag = domesticated_alien_home_crystal } } } } spawn_planet = { class = pc_frozen location = prev has_ring = no init_effect = { set_name = "NAME_CRYSTAL_HOMEWORLD" set_planet_flag = domesticated_alien_home set_planet_flag = domesticated_alien_home_crystal set_planet_size = 40 clear_deposits = yes } } }
					else_if = { limit = { event_target:main_country = { has_country_flag = space_domestication_cloud NOT = { any_planet_within_border = { has_planet_flag = domesticated_alien_home_cloud } } } } spawn_planet = { class = pc_black_hole location = prev has_ring = no init_effect = { set_name = "NAME_CLOUD_HOMEWORLD" set_planet_flag = domesticated_alien_home set_planet_flag = domesticated_alien_home_cloud clear_deposits = yes } } } else_if = { limit = { event_target:main_country = { has_country_flag = space_domestication_drone NOT = { any_planet_within_border = { has_planet_flag = domesticated_alien_home_drone } } } } spawn_planet = { class = pc_asteroid location = prev has_ring = no init_effect = { set_name = "NAME_DRONE_HOMEWORLD" set_planet_flag = domesticated_alien_home set_planet_flag = domesticated_alien_home_drone set_planet_size = 25 clear_deposits = yes } } }
				}
			}
		}
	}
}

	random_relation = {

		limit = { is_hostile = event_target:main_country NOT = { is_hostile = event_target:domesticated_spaceborn_alien } }
		set_faction_hostility = { target = event_target:domesticated_spaceborn_alien set_hostile = yes set_neutral = no set_friendly = no }

		limit = { NOT = { is_hostile = event_target:main_country } is_hostile = event_target:domesticated_spaceborn_alien }
		set_faction_hostility = { target = event_target:domesticated_spaceborn_alien set_hostile = no set_neutral = yes set_friendly = no }
	}
}

	if = {
		limit = { any_country = { has_country_flag = leviathan_master_country } }
		random_country = {
			limit = { has_country_flag = leviathan_master_country }
			save_event_target_as = leviathan_master_country
		}
	} else = {
		create_country = {
			name = "Leviathan Master"
			type = leviathan_master
			flag = {
				icon = { category = "zoological" file = "flag_zoological_5.dds" }

			}
			effect = {
				save_event_target_as = leviathan_master_country
				set_country_flag = leviathan_master_country
				every_country = { establish_communications_no_message = prev }
			}
		}
	}
}
