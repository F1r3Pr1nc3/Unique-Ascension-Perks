# Returns fallback capital_planet
# This = country
get_capital_planet = {
	if = {
		limit = {
			NOT = { exists = event_target:capital_planet }
		}
		if = {
			limit = { exists = capital_scope }
			capital_scope = { save_event_target_as = capital_planet }
		}
		else_if = {
			limit = {
				num_owned_planets > 0
				any_owned_planet = { is_colony = yes }
			}
			if = {
				limit = {
					num_sectors > 0
					any_owned_sector = { exists = sector_capital }
				}
				ordered_owned_sector = {
					limit = { exists = sector_capital }
					position = 0
					order_by = trigger:pop_amount
					sector_capital = { save_event_target_as = capital_planet }
				}
			}
			else = {
				ordered_owned_planet = {
					limit = {
						is_colony = yes
						is_colonizable = no
						exists = space_owner
						solar_system = { exists = starbase }
					}
					position = 0
					order_by = trigger:sapient_pop_amount
					save_event_target_as = capital_planet
				}
			}
			if = {
				limit = {
					NOT = { exists = event_target:capital_planet }
				}
				ordered_planet_within_border = {
					limit = {
						has_owner = yes
						is_colony = yes
						OR = {
							AND = {
								exists = owner
								is_owned_by = prev
							}
							AND = {
								exists = controller
								controller = { is_same_empire = prev }
							}
						}
					}
					position = 0
					order_by = trigger:sapient_pop_amount
					save_event_target_as = capital_planet
				}
			}
			else = {
				random_owned_planet = {
					limit = {
						has_owner = yes
						pop_amount > 0
					}
					weights = {
						base = 1
						modifier = {
							add = 4
							solar_system = { is_capital_system = yes }
						}
						modifier = { add = 1 pop_amount > 500 }
						modifier = { add = 3 pop_amount > 3000 }
					}
					save_event_target_as = capital_planet
				}
				log = "FAIL: No planet found for [This.GetName]?!"
			}
		}
		else = {
			random_owned_planet = {
				limit = {
					has_owner = yes
					pop_amount > 0
				}
				weights = {
					base = 1
					modifier = {
						add = 4
						solar_system = { is_capital_system = yes }
					}
					modifier = { add = 1 pop_amount > 500 }
					modifier = { add = 3 pop_amount > 3000 }
				}
				save_event_target_as = capital_planet
			}
			log = "FAIL: No planet found for [This.GetName]?!"
		}
	}
}
