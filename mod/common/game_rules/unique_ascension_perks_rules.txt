# This = country
can_support_branch_offices = {
	custom_tooltip = {
		fail_text = BRANCH_OFFICE_NOT_FALLEN_EMPIRES
		is_fallen_empire = no
	}
	custom_tooltip = {
		fail_text = BRANCH_OFFICE_NOT_ON_MEGACORP_PLANET
		NOR = {
			is_megacorp = yes			# has_authority = auth_corporate
			has_valid_civic = civic_galactic_sovereign_megacorp
		}
	}
	custom_tooltip = {
		fail_text = BRANCH_OFFICE_NOT_REGULAR_EMPIRE
		is_regular_empire = yes
	}
}

# This = planet
# Root = country
can_build_branch_office_on_planet = {
	custom_tooltip = {
		fail_text = requires_recipient_not_fanatic_purifiers
		exists = owner
		owner = {
			OR = {
				is_same_species = root
				NOT = { has_valid_civic = civic_fanatic_purifiers }
			}
		}
	}
	custom_tooltip = {
		fail_text = BRANCH_OFFICE_NOT_COMMERCIAL_PACT
		exists = owner
		OR = {
			AND = {
				owner = { merg_is_default_empire = yes }
				# NOT = { root = { has_authority = auth_corporate } }
				OR = {
					root = { is_criminal_syndicate = yes }
					owner = { has_commercial_pact = root }
					owner = { is_in_federation_with = root }
					# Allow branch offices in subsidiaries and vice versa.
					root = {
						is_subject = yes
						# any_agreement = { agreement_preset = preset_subsidiary }
						OR = {
							has_authority = auth_corporate
							has_valid_civic = civic_galactic_sovereign_megacorp
						}
						overlord = {
							OR = {
								is_same_empire = prevprev.owner
								AND = {
									exists = prevprev.owner.overlord
									is_same_empire = prevprev.owner.overlord
								}
							}
						}
					}
					owner = {
						is_subject = yes
						# any_agreement = { agreement_preset = preset_subsidiary }
						overlord = {
							OR = {
								is_same_empire = root
								is_same_empire = root.overlord
							}
							OR = {
								has_authority = auth_corporate
								has_valid_civic = civic_galactic_sovereign_megacorp
							}
						}
					}
					AND = {
						root = { has_valid_civic = civic_galactic_sovereign_megacorp }
						owner = { is_galactic_community_member = yes }
					}
					AND = {
						root = { has_modifier = imperial_charter }
						owner = { is_galactic_community_member = yes }
					}
					# Imposed Megacorp allows branch offices without commercial (MOD)
					root = { has_ascension_perk = ap_megacorp }
				}
			}
			# Imposed Megacorp allows branch offices on Gesalt Empire (MOD)
			AND = {
				owner = { is_gestalt = yes }
				root = { has_ascension_perk = ap_megacorp }
			}
			# Imposed Megacorp allows branch offices on other MegaCorp (MOD)
			# AND = { owner = { has_authority = auth_corporate } root = { has_ascension_perk = ap_megacorp } }
		}
	}
	custom_tooltip = {
		fail_text = BRANCH_OFFICE_MEGACORP_OVERLORD
		exists = owner
		# Failure message for having a corporate overlord.
		if = {
			limit = {
				OR = {
					# Not valid if you aren't a subject.
					root = { is_subject = no }
					# Or if your overlord isn't a megacorp.
					root = {
						is_subject = yes
						overlord = {
							NOR = {
								has_authority = auth_corporate
								has_valid_civic = civic_galactic_sovereign_megacorp
							}
						}
					}
				}
			}
			always = yes
		} else = {
			AND = {
				root = {
					is_subject = yes
					overlord = {
						OR = {
							has_authority = auth_corporate
							has_valid_civic = civic_galactic_sovereign_megacorp
						}
					}
				}
				owner = {
					OR = {
						is_subject = no
						AND = {
							is_subject = yes
							overlord = { NOT = { is_same_empire = root.overlord } }
						}
					}
				}
			}
		}
	}
	custom_tooltip = {
		fail_text = BRANCH_OFFICE_CRIMINAL_HERITAGE_HAS_TRUCE
		exists = owner
		OR = {
			root = { is_criminal_syndicate = no }
			owner = {
				NOR = {
					has_truce = root
					is_at_war_with = root
				}
			}
		}
	}
	custom_tooltip = {
		fail_text = BRANCH_OFFICE_CRIMINAL_COOLDOWN
		exists = owner
		OR = {
			root = { is_criminal_syndicate = no }
			AND = {
				root = { is_criminal_syndicate = yes }
				NOT = { has_planet_flag = criminal_branch_closed_cooldown_planet@root }
			}
		}
	}
}
